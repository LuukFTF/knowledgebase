controller

class Products {
  function getProduct(sku){
    product = this.ProductsModel.getProduct(sku)

    renderView('views/products', product)
  }
}

model

class ProductsModel {
  function getProduct(sku) {
    data = this.db.get('SELECT * FROM products WHERE sku = sku')
    return data
  }
}
